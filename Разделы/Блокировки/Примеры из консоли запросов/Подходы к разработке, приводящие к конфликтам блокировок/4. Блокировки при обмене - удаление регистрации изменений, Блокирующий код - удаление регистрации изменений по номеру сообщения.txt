
Узел = ПланыОбмена.Документы.НайтиПоКоду("Выгружено");
НомерПринятогоСобщения = 3;

Предупреждение("Начало удаления регистрации изменений для сообщения № "+НомерПринятогоСобщения);

// при подтверждении приема сообщения как правило удаляют регистрацию изменений 
// для номера сообщения <= номер принятого сообщения метод блокирует
// для каждого объекта метаданных изменения, которые удаляются
//
// Пример: удалили регистрацию для 1000 изменений Документ1 = заблокировали 1000 ссылок Документ1
ПланыОбмена.УдалитьРегистрациюИзменений(Узел, НомерПринятогоСобщения);

Предупреждение("Окончание удаления регистрации изменений для сообщения № "+НомерПринятогоСобщения);
